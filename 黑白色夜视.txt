#include <Wire.h>
#include <Adafruit_MLX90640.h>
#include <SPI.h>
#include <Adafruit_GFX.h>
#include <Adafruit_ILI9341.h>

// ===== ILI9341 引脚 =====
#define TFT_CS   7
#define TFT_DC   6
#define TFT_RST  8
#define TFT_MOSI 17
#define TFT_SCLK 18
#define TFT_MISO -1

Adafruit_ILI9341 tft = Adafruit_ILI9341(TFT_CS, TFT_DC, TFT_RST);

// ===== MLX90640 引脚 =====
#define SDA_PIN 3
#define SCL_PIN 4
Adafruit_MLX90640 mlx;
float frame[32*24];

// ===== 横屏分辨率 =====
const int screenWidth  = 320;
const int screenHeight = 240;

// ===== 原始热像素 =====
const int pixelCols = 32;
const int pixelRows = 24;

// ===== 夜视效果颜色映射：热 = 白，冷 = 黑 =====
uint16_t colorMap(float t, float tmin, float tmax) {
  float f = (t - tmin) / (tmax - tmin);
  f = constrain(f, 0.0, 1.0);

  // 将温度映射到 0~255 的灰度值
  uint8_t gray = (uint8_t)(f * 255);  
  return tft.color565(gray, gray, gray);
}

void setup() {
  Serial.begin(115200);

  // 初始化屏幕
  SPI.begin(TFT_SCLK, TFT_MISO, TFT_MOSI);
  tft.begin();
  tft.setRotation(3);   // 横屏
  tft.fillScreen(ILI9341_BLACK);

  // 初始化 MLX90640
  Wire.begin(SDA_PIN, SCL_PIN);
  if(!mlx.begin(0x33,&Wire)){
    Serial.println("未检测到 MLX90640！");
    while(1);
  }
  mlx.setMode(MLX90640_INTERLEAVED);
//  mlx.setMode(MLX90640_CHESS);

  mlx.setRefreshRate(MLX90640_4_HZ);
  Serial.println("MLX90640 初始化成功");
}

void loop() {
  if(mlx.getFrame(frame) != 0){
    Serial.println("读取失败");
    return;
  }

  // 找最小最大温度
  float tmin = 1000, tmax = -1000;
  for(int i=0;i<pixelCols*pixelRows;i++){
    if(frame[i]<tmin) tmin=frame[i];
    if(frame[i]>tmax) tmax=frame[i];
  }
  float range = tmax - tmin;
  if(range<0.1) range=0.1;

  // === 横屏平滑插值绘制 ===
  for(int y=0; y<screenHeight; y++){
    float fy = y * (pixelRows-1.0) / (screenHeight-1);
    int y0 = (int)fy;
    int y1 = min(y0+1,pixelRows-1);
    float dy = fy - y0;

    for(int x=0; x<screenWidth; x++){
      float fx = x * (pixelCols-1.0) / (screenWidth-1);
      int x0 = (int)fx;
      int x1 = min(x0+1,pixelCols-1);
      float dx = fx - x0;

      // 双线性插值
      float t00 = frame[y0*pixelCols + x0];
      float t10 = frame[y0*pixelCols + x1];
      float t01 = frame[y1*pixelCols + x0];
      float t11 = frame[y1*pixelCols + x1];
      float temp = t00*(1-dx)*(1-dy) + t10*dx*(1-dy) + t01*(1-dx)*dy + t11*dx*dy;

      uint16_t color = colorMap(temp, tmin, tmax);
      tft.drawPixel(x, y, color);
    }
  }

  delay(100); // 控制刷新
}
